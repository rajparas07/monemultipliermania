    <script>
        // Audio resources
        const sounds = {
            click: new Audio('https://cdn.jsdelivr.net/gh/themaximal1st/free-web-sounds@main/UISounds/click.mp3'),
            success: new Audio('https://cdn.jsdelivr.net/gh/themaximal1st/free-web-sounds@main/UISounds/success.mp3'),
            moneySound: new Audio('https://cdn.jsdelivr.net/gh/themaximal1st/free-web-sounds@main/UISounds/coin.mp3')
        };
        
        // Game Variables
        let moneyValue = 100;
        let clickCount = 0;
        let currencySymbols = {
            USD: '$',
            EUR: '',
            GBP: '',
            JPY: '',
            CAD: 'C$',
            AUD: 'A$',
            CNY: '',
            INR: ''
        };
        
        // DOM Elements
        const gameArea = document.getElementById('gameArea');
        const startGameBtn = document.getElementById('startGameBtn');
        const multiplyBtn = document.getElementById('multiplyBtn');
        const resetBtn = document.getElementById('resetBtn');
        const helpBtn = document.getElementById('helpBtn');
        const closeHelpBtn = document.getElementById('closeHelpBtn');
        const helpModal = document.getElementById('helpModal');
        const soundToggle = document.getElementById('soundToggle');
        const moneyValueEl = document.getElementById('moneyValue');
        const currencySymbolEl = document.getElementById('currencySymbol');
        const currencySelect = document.getElementById('currency');
        const clickCountEl = document.getElementById('clickCount');
        const multiplierLogEl = document.getElementById('multiplierLog');
        const moneyContainer = document.querySelector('.money-container');
        
        // Event Listeners
        startGameBtn.addEventListener('click', startGame);
        multiplyBtn.addEventListener('click', multiplyMoney);
        resetBtn.addEventListener('click', resetGame);
        helpBtn.addEventListener('click', showHelp);
        closeHelpBtn.addEventListener('click', hideHelp);
        currencySelect.addEventListener('change', updateCurrencySymbol);
        
        // Initialize game
        updateCurrencySymbol();
        
        // Functions
        function startGame() {
            if (soundToggle.checked) {
                sounds.click.play();
            }
            gameArea.classList.remove('hidden');
            startGameBtn.disabled = true;
            startGameBtn.classList.add('opacity-50', 'cursor-not-allowed');
            
            // Add player to leaderboard
            addPlayerToLeaderboard();
            
            // Scroll to game area
            gameArea.scrollIntoView({ behavior: 'smooth' });
        }
        
        function updateCurrencySymbol() {
            const selectedCurrency = currencySelect.value;
            currencySymbolEl.textContent = currencySymbols[selectedCurrency];
        }
        
        function multiplyMoney() {
            if (soundToggle.checked) {
                sounds.click.play();
            }
            
            clickCount++;
            clickCountEl.textContent = `Clicks: ${clickCount}`;
            
            // Random multiplier between 1.1 and 5
            let multiplier = 1 + Math.random() * 4;
            
            // Chance for special multiplier boost (10% chance)
            if (Math.random() < 0.1) {
                multiplier = 5 + Math.random() * 10;
                if (soundToggle.checked) {
                    sounds.success.play();
                }
                multiplierLogEl.innerHTML = `Last boost: <span class="text-green-600 font-bold">SUPER BOOST x${multiplier.toFixed(2)}!</span>`;
            } else {
                multiplierLogEl.textContent = `Last boost: x${multiplier.toFixed(2)}`;
            }
            
            // Update money value
            const oldValue = moneyValue;
            moneyValue = Math.floor(moneyValue * multiplier);
            moneyValueEl.textContent = formatMoney(moneyValue);
            
            // Add visual effects
            createMoneyAnimation();
            createSparkleEffect();
            pulseEffect(moneyValueEl);
            
            // Trigger confetti on big multipliers
            if (multiplier > 3) {
                triggerConfetti();
            }
            
            // Update leaderboard
            updateLeaderboard();
        }
        
        function formatMoney(amount) {
            return amount.toLocaleString();
        }
        
        function resetGame() {
            if (soundToggle.checked) {
                sounds.click.play();
            }
            
            moneyValue = 100;
            clickCount = 0;
            moneyValueEl.textContent = '100';
            clickCountEl.textContent = 'Clicks: 0';
            multiplierLogEl.textContent = 'Last boost: None';
            updateLeaderboard();
        }
        
        function showHelp() {
            if (soundToggle.checked) {
                sounds.click.play();
            }
            helpModal.classList.remove('hidden');
        }
        
        function hideHelp() {
            if (soundToggle.checked) {
                sounds.click.play();
            }
            helpModal.classList.add('hidden');
        }
        
        function createMoneyAnimation() {
            if (soundToggle.checked) {
                sounds.moneySound.currentTime = 0;
                sounds.moneySound.play();
            }
            
            // Create 5-15 floating money bills
            const numberOfBills = 5 + Math.floor(Math.random() * 10);
            for (let i = 0; i < numberOfBills; i++) {
                const bill = document.createElement('div');
                bill.className = 'floating-money';
                bill.textContent = currencySymbolEl.textContent;
                
                // Random position and delay
                const leftPos = Math.random() * 100;
                const delay = Math.random() * 0.5;
                
                bill.style.left = `${leftPos}%`;
                bill.style.animationDelay = `${delay}s`;
                
                moneyContainer.appendChild(bill);
                
                // Remove bill after animation
                setTimeout(() => {
                    bill.remove();
                }, 3000 + (delay * 1000));
            }
        }
        
        function createSparkleEffect() {
            // Create 3-8 sparkles
            const numberOfSparkles = 3 + Math.floor(Math.random() * 5);
            for (let i = 0; i < numberOfSparkles; i++) {
                const sparkle = document.createElement('div');
                sparkle.className = 'sparkle';
                
                // Random position and delay
                const leftPos = 20 + Math.random() * 60;
                const topPos = 20 + Math.random() * 60;
                const delay = Math.random() * 0.3;
                
                sparkle.style.left = `${leftPos}%`;
                sparkle.style.top = `${topPos}%`;
                sparkle.style.animationDelay = `${delay}s`;
                
                moneyContainer.appendChild(sparkle);
                
                // Remove sparkle after animation
                setTimeout(() => {
                    sparkle.remove();
                }, 2000 + (delay * 1000));
            }
        }
        
        function pulseEffect(element) {
            element.classList.add('pulse');
            setTimeout(() => {
                element.classList.remove('pulse');
            }, 500);
        }
        
        function triggerConfetti() {
            confetti({
                particleCount: 100,
                spread: 70,
                origin: { y: 0.6 }
            });
        }
        
        function addPlayerToLeaderboard() {
            const leaderboard = document.getElementById('leaderboard');
            const playerItem = document.createElement('li');
            playerItem.className = 'leaderboard-item p-2 rounded-lg bg-pink-100 flex justify-between';
            playerItem.id = 'playerScore';
            playerItem.innerHTML = `
                <span><i class="fas fa-user text-pink-500 mr-2"></i>You</span>
                <span>${currencySymbolEl.textContent}${formatMoney(moneyValue)}</span>
            `;
            leaderboard.appendChild(playerItem);
        }
        
        function updateLeaderboard() {
            const playerScore = document.getElementById('playerScore');
            if (playerScore) {
                playerScore.querySelector('span:last-child').textContent = `${currencySymbolEl.textContent}${formatMoney(moneyValue)}`;
                
                // Sort the leaderboard
                const leaderboard = document.getElementById('leaderboard');
                const items = Array.from(leaderboard.children);
                
                // Move player based on score
                let playerIndex = items.indexOf(playerScore);
                for (let i = 0; i < items.length; i++) {
                    if (i === playerIndex) continue;
                    
                    const scoreText = items[i].querySelector('span:last-child').textContent;
                    const score = parseInt(scoreText.replace(/[^0-9]/g, ''));
                    
                    if (moneyValue > score && i < playerIndex) {
                        leaderboard.insertBefore(playerScore, items[i]);
                        break;
                    } else if (moneyValue < score && i > playerIndex) {
                        leaderboard.insertBefore(playerScore, items[i].nextSibling);
                        break;
                    }
                }
                
                // Highlight player position
                playerScore.classList.add('bg-pink-100');
                if (playerIndex === 0) {
                    triggerConfetti();
                    if (soundToggle.checked) {
                        sounds.success.play();
                    }
                }
            }
        }
        
        // Listen for sound toggle changes
        soundToggle.addEventListener('change', () => {
            // No need to do anything else, the checked state will be used when playing sounds
        });
        
        // Close help modal when clicking outside
        window.addEventListener('click', (event) => {
            if (event.target === helpModal) {
                hideHelp();
            }
        });
    </script>
